  <script>
    const form = document.getElementById('contactForm');
    const errorMsg = document.getElementById('errorMsg');
    const enderecoInput = document.getElementById('endereco');

    form.addEventListener('submit', function(event) {
      errorMsg.textContent = '';
      let valid = true;

      const telefone = document.getElementById('telefone').value.trim();
      const instagram = document.getElementById('instagram').value.trim();
      const email = document.getElementById('email').value.trim();

      const telefoneRegex = /^\+\d{1,3}\s\(\d{1,3}\)\s\d{4,5}-\d{4}$/;
      if (telefone && !telefoneRegex.test(telefone)) {
        valid = false;
        errorMsg.textContent += 'Telefone inválido. Formato: +55 (11) 91234-5678\n';
      }

      if (instagram && !/^@/.test(instagram)) {
        valid = false;
        errorMsg.textContent += 'Instagram inválido. Deve começar com @\n';
      }

      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (email && !emailRegex.test(email)) {
        valid = false;
        errorMsg.textContent += 'Email inválido.\n';
      }

      if (!valid) {
        event.preventDefault();
      }
    });

    // Função do Google Maps
    let map;
    let geocoder;
    function initMap() {
      geocoder = new google.maps.Geocoder();
      map = new google.maps.Map(document.getElementById("map"), {
        zoom: 4,
        center: { lat: -15.7942, lng: -47.8822 } // Brasil como centro inicial
      });
    }

    // Atualiza mapa ao digitar endereço
    enderecoInput.addEventListener('change', () => {
      const address = enderecoInput.value;
      if (!address) return;

      geocoder.geocode({ address: address }, (results, status) => {
        if (status === "OK") {
          map.setCenter(results[0].geometry.location);
          new google.maps.Marker({
            map: map,
            position: results[0].geometry.location
          });
        } else {
          alert("Endereço não encontrado: " + status);
        }
      });
    });
  </script>
